async function uploadAndVerify() { const fileInput = document.getElementById('slipInput'); const statusDiv = document.getElementById('status'); // แก้ไขปัญหา ReferenceError const file = fileInput.files[0]; if (!file) { alert("กรุณาเลือกไฟล์สลิปก่อนครับ"); return; } statusDiv.style.color = "blue"; statusDiv.innerText = "⏳ กำลังตรวจสอบสลิป..."; const formData = new FormData(); formData.append('slip', file); // ชื่อ 'slip' ต้องตรงกับใน server.js try { const response = await fetch('/verify-slip', { method: 'POST', body: formData }); const result = await response.json(); if (response.ok && result.success) { statusDiv.style.color = "green"; statusDiv.innerHTML = `✅ ยืนยันสำเร็จ! ยอดโอน: ${result.data.amount} บาท`; } else { statusDiv.style.color = "red"; statusDiv.innerText = `❌ ${result.message || 'ตรวจสอบไม่สำเร็จ'}`; } } catch (error) { statusDiv.style.color = "red"; statusDiv.innerText = "⚠️ ไม่สามารถติดต่อเซิร์ฟเวอร์ได้"; } }
